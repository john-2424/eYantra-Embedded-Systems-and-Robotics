118

125  wl_r_sensors_port_reg |= (1<<right_wl_sensor_pin) ;

164 /*
165  * Function Name: timer_pwm_init
166  * Input: None
167  * Output: None
168  * Logic: Function to initialize Timer 0 and 2 in 8-bit Fast PWM mode for speed control of motors
169  * Example Call: timer_pwm_init();
170  */
171 void timer_pwm_init(void)
172 {
173 
174  // To configure Compare Output Mode bits to select Clear on Compare setting in TCCRnA register
175  // To configure Waveform Generation Mode bits for 8-bit Fast PWM in TCCRnA and TCCRnB register
176  // For Timer 0
177  TCCR0A_reg |= ((1<<WGM0_bit) | (1<<COMB1_bit));
178  TCCR0A_reg &= ~((1<<WGM1_bit) | (1<<COMB0_bit));
179 
180  // For Timer 2
181  TCCR2A_reg |= ((1<<WGM0_bit) | (1<<COMB1_bit));
182  TCCR2A_reg &= ~((1<<WGM1_bit) | (1<<COMB0_bit));
183 
184  // To configure Clock Select bits to choose prescalar division factor of 64
185  // For Timer 0
186  TCCR0B_reg |= ((1<<CS0_bit) | (1<<CS1_bit));
187  TCCR0B_reg &= ~((1<<CS2_bit) | (1<<WGM2_bit));
188 
189  // For Timer 2
190  TCCR2B_reg |= ((1<<CS2_bit));
191  TCCR2B_reg &= ~((1<<CS1_bit) | (1<<CS0_bit) | (1<<WGM2_bit));
192 
193  // To make values in both Output Compare Registers as 0
194  OCR0B_reg = 0x00;
195  OCR2B_reg = 0x00;
196 }

301  left_status = true;
302  else

306  center_status = true;
307  else
